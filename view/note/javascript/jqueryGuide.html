  • selector

    왼쪽은 덜 세밀하게 오른쪽은 세밀하게
    							$(".data td.gonzalez");
    ID > child 의 구조로 엘레먼트를 접근하는 경우에는 나열하지말고 해당 엘레먼트를 find() 함수를 사용하여 접근
    							var $productIds = $("#products div.id");
        					var $productIds = $("#products").find("div.id"); //good
    과도하게 세밀한 접근은 피하기
    							$(".data table.attendees td.gonzalez");
         					$(".data td.gonzalez"); //good
    Context를 함께 제공
    							$('.class');
        					$('.class', '#class-container'); //good
    보이지 않는 Universal 한 셀렉팅을 피하기
    							$('div.someclass :radio');
        					$('div.someclass input:radio'); //good

  • DOM manipulation

    array.join() 사용
    							// BAD
					        var $myList = $("#list");
					        for(var i = 0; i < 10000; i++){
					            $myList.append("
  • "+i+"
    • ");
					        }
					        
					        // GOOD
					        var $myList = $("#list");
					        var list = "";
					        for(var i = 0; i < 10000; i++){
					            list += "
  • "+i+"
    • ";
					        }
					        $myList.html(list);
					        
					        // EVEN FASTER
					        var array = []; 
					        for(var i = 0; i < 10000; i++){
					            array[i] = "
  • "+i+"
    • "; 
					        }
					        $myList.html(array.join(''));
